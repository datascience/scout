PREFIX xsd: <http://www.w3.org/2001/XMLSchema#>
PREFIX rdfs: <http://www.w3.org/2000/01/rdf-schema#>
PREFIX rdf: <http://www.w3.org/1999/02/22-rdf-syntax-ns#>
PREFIX owl: <http://www.w3.org/2002/07/owl#>
PREFIX pw: <http://scape-project.eu/pw/vocab/>
PREFIX cp: <http://scape-project.eu/pw/vocab/control-policy#>

SELECT ?objective ?measure ?name ?desc ?modality ?value ?qualifier WHERE { 
  ?objectiveSubclass rdfs:subClassOf cp:Objective .
  ?objective a ?objectiveSubclass ;
             cp:property ?measure;
             cp:modality ?modality ;
             cp:value ?value .
  OPTIONAL {
    ?objective cp:qualifier ?qualifier .
  }
  ?measure a pw:Measure .
  OPTIONAL {
    ?measure pw:description ?desc ;
             rdfs:label ?name .
  }
}
